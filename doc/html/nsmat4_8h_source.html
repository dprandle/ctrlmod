431  nsmat4<T> & rotationFrom(const NSVec3<T> & vec, const NSVec3<T> & toVec)
432  {
433  /* http://www.euclideanspace.com/maths/algebra/vectors/angleBetween/index.htm */
434  NSVec3<T> vs = cross(vec, toVec);
435  T ca = vec * toVec;
436  NSVec3<T> v = normalize(vs);
437  NSVec3<T> vt = v * (static_cast<T>(1) - ca);
438 
439  data[0][0] = vt.x * v.x + ca;
440  data[1][1] = vt.y * v.y + ca;
441  data[2][2] = vt.z * v.z + ca;
442  vt.x *= v.y;
443  vt.z *= v.x;
444  vt.y *= v.z;
445  data[0][1] = vt.x - vs.z;
446  data[0][2] = vt.z + vs.y;
447  data[1][0] = vt.x + vs.z;
448  data[1][2] = vt.y - vs.x;
449  data[2][0] = vt.z - vs.y;
450  data[2][1] = vt.y + vs.x;
451 
452  // Set remaining components
453  data[0][3] = static_cast<T>(0); data[1][3] = static_cast<T>(0); data[2][3] = static_cast<T>(0);
454  data[3][0] = static_cast<T>(0); data[3][1] = static_cast<T>(0); data[3][2] = static_cast<T>(0); data[3][3] = static_cast<T>(1);
455  return *this;
