211 T edpid_controller<T>::loop(const T & input, double dt)
212 {
213  T current_error = m_target - input;
214 
215  // Limit the error if an output range is set (default ramping is one, or the entire range)
216  if (m_range.min != m_range.max)
217  {
218  T max_err = (m_range.max - m_range.min) * m_ramp_limit;
219  if (current_error > max_err)
220  current_error = max_err;
221  }
222 
223  // Integral portion
224  m_integral_err += current_error * dt;
225 
226  // Derivitive portion
227  T deriv;
228  if (m_complex_deriv)
229  deriv = (current_error + 3.0 * (m_prev_error[2] - m_prev_error[1]) - m_prev_error[0]) / 6.0;
230  else
231  deriv = (current_error - m_prev_error[2]) / dt;
232 
233  // Shift errors to the left and set last error
234  for (uint8_t i = 0; i < 2; ++i)
235  m_prev_error[i] = m_prev_error[i+1];
236  m_prev_error[2] = current_error;
237 
238  T ret = (m_gain.P * current_error + m_gain.I * m_integral_err + m_gain.D * deriv);
239 
240  // limit the output if a valid range is set
241  // also, if anti-reset-windup is set then undo the earlier summation to the integral term
242  if (m_range.min != m_range.max)
243  {
244  if (ret > m_range.max)
245  {
246  ret = m_range.max;
247  if (m_anti_reset_windup)
248  m_integral_err -= current_error * dt;
249  }
250  if (ret < m_range.min)
251  {
252  ret = m_range.min;
253  if (m_anti_reset_windup)
254  m_integral_err -= current_error * dt;
255  }
256  }
257 
258  return ret;
259 }
